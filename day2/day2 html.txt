  <script>
    (function(){
      function qs(sel, root){ return (root||document).querySelector(sel); }
      function qsa(sel, root){ return Array.from((root||document).querySelectorAll(sel)); }

      // captions fallback：沒寫 data-caption 就用 img alt
      qsa('.galleryRow .gItem').forEach(function(item){
        var img = item.querySelector('img');
        if(!img) return;
        var cap = (item.getAttribute('data-caption')||'').trim();
        if(cap) return;
        var alt = (img.getAttribute('alt')||'').trim();
        if(alt) item.setAttribute('data-caption', alt);
      });

      // lightbox
      function initLightbox(){
        if(!qs('.lbModal')){
          var lb = document.createElement('div');
          lb.className = 'lbModal';
          lb.innerHTML = ''
            + '<div class="lbBackdrop" data-lb-close></div>'
            + '<div class="lbStage"><img class="lbImg" alt="" /></div>'
            + '<div class="lbNav">'
            + '  <button class="lbBtn" type="button" data-lb-prev aria-label="上一張">‹</button>'
            + '  <button class="lbBtn" type="button" data-lb-next aria-label="下一張">›</button>'
            + '  <span class="lbCount" data-lb-count>0/0</span>'
            + '</div>'
            + '<button class="lbClose" type="button" data-lb-close aria-label="關閉">✕</button>';
          document.body.appendChild(lb);
        }

        var modal = qs('.lbModal');
        var stage = qs('.lbStage', modal);
        var imgEl = qs('.lbImg', modal);
        var countEl = qs('[data-lb-count]', modal);
        var list = [], idx = 0;

        function openAt(i){
          if(!list.length) return;
          idx = (i + list.length) % list.length;
          imgEl.src = list[idx];
          countEl.textContent = String(idx + 1) + '/' + String(list.length);
          modal.classList.add('open');
          document.body.style.overflow = 'hidden';
        }
        function close(){ modal.classList.remove('open'); document.body.style.overflow=''; }
        function next(){ openAt(idx + 1); }
        function prev(){ openAt(idx - 1); }

        modal.addEventListener('click', function(e){
          if(e.target && e.target.closest && e.target.closest('[data-lb-close]')) close();
        });
        qs('[data-lb-next]', modal).addEventListener('click', function(e){ e.stopPropagation(); next(); });
        qs('[data-lb-prev]', modal).addEventListener('click', function(e){ e.stopPropagation(); prev(); });

        window.addEventListener('keydown', function(e){
          if(!modal.classList.contains('open')) return;
          if(e.key === 'Escape') close();
          if(e.key === 'ArrowRight') next();
          if(e.key === 'ArrowLeft') prev();
        });

        document.addEventListener('click', function(e){
          var thumb = e.target && e.target.closest && e.target.closest('.galleryRow .gItem img');
          if(!thumb) return;
          if(thumb.getAttribute('data-no-lightbox') === '1') return; // PDF 封面照常開 PDF

          var row = thumb.closest('.galleryRow');
          var imgs = qsa('.gItem img', row);
          list = imgs.map(function(x){ return x.getAttribute('data-full') || x.currentSrc || x.src; });
          openAt(imgs.indexOf(thumb));
        }, { passive:true });

        // swipe
        var tsx=0, tsy=0, tlx=0, tly=0, tracking=false, TH=50;
        stage.addEventListener('touchstart', function(e){
          if(!modal.classList.contains('open')) return;
          if(!e.touches || e.touches.length !== 1) return;
          var t = e.touches[0];
          tsx = tlx = t.clientX; tsy = tly = t.clientY; tracking = true;
        }, { passive:true });
        stage.addEventListener('touchmove', function(e){
          if(!tracking || !e.touches || e.touches.length !== 1) return;
          var t = e.touches[0]; tlx = t.clientX; tly = t.clientY;
        }, { passive:true });
        stage.addEventListener('touchend', function(){
          if(!tracking) return;
          var dx = tlx - tsx, dy = tly - tsy;
          if(Math.abs(dx) > TH && Math.abs(dx) > Math.abs(dy) * 1.2){
            if(dx < 0) next(); else prev();
          }
          tracking = false;
        }, { passive:true });
      }

      window.addEventListener('DOMContentLoaded', function(){
        try{ initLightbox(); }catch(e){ console.warn('[Day2] lightbox init error', e); }
      });
    })();
